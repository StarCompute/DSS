<script>
// 搜索防抖 300ms
function debounce(fn, delay = 300) {
    let timer;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => fn(...args), delay);
    };
}
const searchInput = document.querySelector('.search-input');
searchInput.addEventListener('input', debounce((e) => {
    const keyword = e.target.value.trim();
    console.log('搜索:', keyword);
    // fetchResults(keyword);
}));

// 清除按钮显隐
const clearBtn = document.querySelector('.search-clear');
searchInput.addEventListener('input', () => {
    clearBtn.style.display = searchInput.value ? 'flex' : 'none';
});
clearBtn.addEventListener('click', () => {
    searchInput.value = '';
    clearBtn.style.display = 'none';
    searchInput.focus();
});
</script>

📄 HTML + CSS
<!-- 高级搜索面板 -->
<div class="full-filter-panel">
    <div class="filter-row">
        <span class="filter-label">关键词</span>
        <div class="search-bar" style="width:280px;">
            <span class="search-icon">🔍</span>
            <input class="search-input" placeholder="搜索名称/描述…">
        </div>
    </div>
    <div class="filter-row">
        <span class="filter-label">条件组</span>
        <select class="filter-select">
            <option>名称</option>
            <option>ID</option>
            <option>标签</option>
        </select>
        <select class="filter-select">
            <option>包含</option>
            <option>等于</option>
            <option>开头是</option>
        </select>
        <input class="filter-value-input" placeholder="输入值…">
        <button class="add-condition">＋</button>
    </div>
    <div class="filter-row" style="justify-content:flex-end;">
        <button class="btn btn-secondary">展开更多条件</button>
        <button class="btn btn-primary">🔍 搜索</button>
    </div>
</div>

<script>
// 高级搜索条件构建
class AdvancedSearch {
    constructor() {
        this.conditions = []; // [{ field, operator, value }]
    }
    addCondition(field, operator, value) {
        this.conditions.push({ field, operator, value });
    }
    removeCondition(index) {
        this.conditions.splice(index, 1);
    }
    buildQuery() {
        return {
            keyword: document.querySelector('.search-input')?.value || '',
            filters: this.conditions
        };
    }
    search() {
        const query = this.buildQuery();
        console.log('高级搜索:', JSON.stringify(query, null, 2));
        // fetchResults(query);
    }
    reset() {
        this.conditions = [];
        document.querySelector('.search-input').value = '';
    }
}
const advSearch = new AdvancedSearch();
advSearch.addCondition('名称', '包含', '设计');
advSearch.search();
</script>

📌 七、搜索 & 筛选设计原则

  • 搜索框默认宽度 320px，自适应内容区宽度
  • 筛选条件不超过 5 个，超过则折叠为"高级筛选"
  • 标签过滤适合 5~8 个选项的场景
  • 搜索结果实时更新，防抖 300ms
  • 已选筛选条件使用可移除标签展示，一目了然
  • 排序状态用箭头方向明确标识升序/降序
  • 筛选后有结果数展示，空结果引导调整条件
