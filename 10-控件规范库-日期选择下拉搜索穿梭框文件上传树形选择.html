// === 可搜索下拉选择器（纯 JS，零依赖，可商用） ===
// 数据获取方式：调用 getValue() → { value, text }
// 数据变化时：仅更新 className 和 display，不替换 DOM，界面稳定无闪烁
class SearchableSelect {
    constructor(container) {
        this.container = container;
        this.input = container.querySelector('input');
        this.dropdown = container.querySelector('.searchable-dropdown');
        this.options = [...this.dropdown.querySelectorAll('.option')];
        this.noResult = this.dropdown.querySelector('.no-result');
        this._selectedValue = null;
        this._selectedText = null;
        this._originalTexts = this.options.map(o => o.textContent); // 保存原始文本，避免 innerHTML 丢失
        this._bindEvents();
    }

    _bindEvents() {
        // 聚焦展开
        this.input.addEventListener('focus', () => {
            this.dropdown.classList.add('show');
            this.container.classList.add('open');
        });

        // 输入过滤 + 高亮（只改 display，不改 innerHTML 结构）
        this.input.addEventListener('input', () => {
            const val = this.input.value.toLowerCase();
            let hasResult = false;
            this.options.forEach((opt, i) => {
                const text = this._originalTexts[i].toLowerCase();
                if (!val || text.includes(val)) {
                    opt.style.display = '';
                    hasResult = true;
                    // 用 data-* 存储高亮范围，通过 CSS ::after 或子节点更新
                    if (val && text.includes(val)) {
                        const idx = text.indexOf(val);
                        const before = this._originalTexts[i].substring(0, idx);
                        const match = this._originalTexts[i].substring(idx, idx + val.length);
                        const after = this._originalTexts[i].substring(idx + val.length);
                        // 使用 textContent + span 精确更新，不丢失事件
                        opt.textContent = '';
                        opt.appendChild(document.createTextNode(before));
                        const hl = document.createElement('span');
                        hl.className = 'highlight';
                        hl.textContent = match;
                        opt.appendChild(hl);
                        opt.appendChild(document.createTextNode(after));
                    } else {
                        opt.textContent = this._originalTexts[i];
                    }
                } else {
                    opt.style.display = 'none';
                }
            });
            if (this.noResult) this.noResult.style.display = hasResult ? 'none' : '';
        });

        // 点击选项 — 界面稳定更新
        this.options.forEach((opt, i) => {
            opt.addEventListener('click', () => {
                this.options.forEach(o => o.classList.remove('selected'));
                opt.classList.add('selected');
                this._selectedValue = opt.getAttribute('data-value');
                this._selectedText = this._originalTexts[i];
                // 仅修改 input.value，不重新渲染整个列表
                this.input.value = this._selectedValue;
                this.dropdown.classList.remove('show');
                this.container.classList.remove('open');
            });
        });

        // 键盘导航
        this.input.addEventListener('keydown', (e) => {
            if (e.key === 'ArrowDown' || e.key === 'ArrowUp') {
                e.preventDefault();
                const visible = this.options.filter(o => o.style.display !== 'none');
                const cur = visible.findIndex(o => o.classList.contains('hover'));
                let next = e.key === 'ArrowDown' ? cur + 1 : cur - 1;
                if (next >= visible.length) next = 0;
                if (next < 0) next = visible.length - 1;
                visible.forEach(o => o.classList.remove('hover'));
                if (visible[next]) visible[next].classList.add('hover');
            }
            if (e.key === 'Enter') {
                e.preventDefault();
                const hovered = this.dropdown.querySelector('.option.hover');
                if (hovered) hovered.click();
            }
            if (e.key === 'Escape') {
                this.dropdown.classList.remove('show');
                this.container.classList.remove('open');
            }
        });

        // 点击外部关闭
        document.addEventListener('click', (e) => {
            if (!this.container.contains(e.target)) {
                this.dropdown.classList.remove('show');
                this.container.classList.remove('open');
            }
        });
    }

    /** 获取当前选中值（数据获取接口） */
    getValue() {
        return { value: this._selectedValue, text: this._selectedText };
    }

    /** 程序化设置值（不触发界面闪烁） */
    setValue(value) {
        this.options.forEach((opt, i) => {
            opt.classList.toggle('selected', opt.getAttribute('data-value') === value);
            if (opt.getAttribute('data-value') === value) {
                this._selectedValue = value;
                this._selectedText = this._originalTexts[i];
                this.input.value = value;
            }
        });
    }
}

// === 标签输入控件（纯 JS，零依赖） ===
// 数据获取方式：调用 getTags() → string[]
// 数据变化时：仅操作单个 chip DOM，不重新渲染整个列表，界面稳定无闪烁
class TagInput {
    constructor(wrap) {
        this.wrap = wrap;
        this.input = wrap.querySelector('input');
        this.maxTags = 20;
        this.maxLen = 30;
        this._bindEvents();
    }

    _bindEvents() {
        // 初始化已有标签的删除按钮
        this.wrap.querySelectorAll('.chip .close').forEach(btn => {
            btn.addEventListener('click', () => btn.parentElement.remove());
        });

        this.input.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' || e.key === ',') {
                e.preventDefault();
                this._addTag(this.input.value);
                this.input.value = '';
            }
            if (e.key === 'Backspace' && this.input.value === '') {
                const chips = this.wrap.querySelectorAll('.chip');
                if (chips.length) chips[chips.length - 1].remove();
            }
        });

        this.input.addEventListener('blur', () => {
            if (this.input.value.trim()) {
                this._addTag(this.input.value);
                this.input.value = '';
            }
        });
    }

    _addTag(text) {
        text = text.trim();
        if (!text) return;
        // 去重
        for (let chip of this.wrap.querySelectorAll('.chip')) {
            if (chip.textContent.replace('×','').trim() === text) return;
        }
        // 上限检查
        const chips = this.wrap.querySelectorAll('.chip');
        if (chips.length >= this.maxTags) return;
        // 长度截断
        if (text.length > this.maxLen) text = text.substring(0, this.maxLen);

        const chip = document.createElement('span');
        chip.className = 'chip';
        chip.innerHTML = text + ' <span class="close">×</span>';
        chip.querySelector('.close').addEventListener('click', () => chip.remove());
        this.wrap.insertBefore(chip, this.input);
    }

    /** 获取所有标签值（数据获取接口） */
    getTags() {
        return [...this.wrap.querySelectorAll('.chip')].map(c =>
            c.textContent.replace('×','').trim()
        );
    }

    /** 程序化设置标签列表（不触发界面闪烁） */
    setTags(tags) {
        // 清除现有标签
        this.wrap.querySelectorAll('.chip').forEach(c => c.remove());
        // 逐个添加
        tags.forEach(t => this._addTag(t));
    }
}

// 初始化
const tagInput = new TagInput(document.getElementById('tagInputWrap'));
// 获取所有标签：tagInput.getTags()  →  ['前端','React','TypeScript']
// 设置标签：tagInput.setTags(['Vue','Pinia','TypeScript'])

// === 穿梭框控件（纯 JS，零依赖） ===
// 数据获取方式：调用 getSelectedValues() → string[]
// 数据变化时：仅移动单个 DOM 节点，不重新渲染，界面稳定无闪烁
function initTransfer(leftPanel, rightPanel, btnRight, btnLeft) {
    function moveItems(fromPanel, toPanel) {
        const checked = fromPanel.querySelectorAll('input[type="checkbox"]:checked');
        checked.forEach(cb => {
            const item = cb.parentElement;
            cb.checked = false;
            // 直接移动 DOM 节点，不重建，保持事件绑定和状态
            toPanel.appendChild(item);
        });
        updateCounts();
    }

    function updateCounts() {
        document.getElementById('leftCount').textContent =
            leftPanel.querySelectorAll('.item').length + ' 项';
        document.getElementById('rightCount').textContent =
            rightPanel.querySelectorAll('.item').length + ' 项';
    }

    btnRight.addEventListener('click', () => moveItems(leftPanel, rightPanel));
    btnLeft.addEventListener('click', () => moveItems(rightPanel, leftPanel));

    // 双击快速移动
    [leftPanel, rightPanel].forEach(panel => {
        panel.addEventListener('dblclick', (e) => {
            const item = e.target.closest('.item');
            if (!item) return;
            const targetPanel = panel === leftPanel ? rightPanel : leftPanel;
            item.querySelector('input').checked = false;
            targetPanel.appendChild(item);
            updateCounts();
        });
    });

    /** 获取右侧已选值列表（数据获取接口） */
    function getSelectedValues() {
        return [...rightPanel.querySelectorAll('.item')].map(item => {
            const cb = item.querySelector('input[type="checkbox"]');
            return cb ? cb.value : '';
        }).filter(Boolean);
    }

    return { getSelectedValues };
}

// 初始化
const transfer = initTransfer(
    document.getElementById('leftPanel'),
    document.getElementById('rightPanel'),
    document.getElementById('btnRight'),
    document.getElementById('btnLeft')
);
// 获取已选值：transfer.getSelectedValues()  →  ['liujiu','zhoushi']

// === 文件上传控件（拖拽 + 点击，纯 JS） ===
// 数据获取方式：调用 getFiles() → File[]
// 数据变化时：仅操作单个文件项 DOM，不重新渲染整个列表，界面稳定无闪烁
class FileUpload {
    constructor(zone, input, list, maxSize = 10 * 1024 * 1024) {
        this.zone = zone;
        this.input = input;
        this.list = list;
        this.maxSize = maxSize;
        this._fileStore = [];  // 存储原始 File 对象，用于数据获取
        this._bindEvents();
    }

    _bindEvents() {
        // 点击触发
        this.zone.addEventListener('click', () => this.input.click());

        this.input.addEventListener('change', (e) => {
            [...e.target.files].forEach(f => this._addFile(f));
            this.input.value = '';
        });

        // 拖拽
        this.zone.addEventListener('dragover', (e) => {
            e.preventDefault();
            this.zone.classList.add('drag-over');
        });
        this.zone.addEventListener('dragleave', () => {
            this.zone.classList.remove('drag-over');
        });
        this.zone.addEventListener('drop', (e) => {
            e.preventDefault();
            this.zone.classList.remove('drag-over');
            [...e.dataTransfer.files].forEach(f => this._addFile(f));
        });
    }

    _addFile(file) {
        // 大小校验
        if (file.size > this.maxSize) {
            alert('文件 ' + file.name + ' 超过大小限制');
            return;
        }

        this._fileStore.push(file);

        const ext = file.name.split('.').pop().toLowerCase();
        const iconMap = { pdf:'📄', doc:'📝', docx:'📝', xls:'📊', xlsx:'📊',
                         jpg:'🖼️', jpeg:'🖼️', png:'🖼️', gif:'🖼️' };

        const item = document.createElement('div');
        item.className = 'upload-file-item';
        item.dataset.fileId = file.name + '_' + Date.now();
        item.innerHTML = `
            <span class="file-icon">${iconMap[ext] || '📎'}</span>
            <div class="file-info">
                <div class="file-name">${file.name}</div>
                <div class="file-size">${(file.size/1024).toFixed(1)} KB · 上传中...</div>
            </div>
            <div class="file-progress"><div class="bar" style="width:0%"></div></div>
            <span class="file-remove">×</span>
        `;
        item.querySelector('.file-remove').addEventListener('click', () => {
            const idx = [...this.list.children].indexOf(item);
            if (idx >= 0) this._fileStore.splice(idx, 1);
            item.remove();
        });
        this.list.appendChild(item);

        // 模拟进度（实际应通过 XHR/Fetch 上传）
        this._simulateProgress(item);
    }

    _simulateProgress(item) {
        const bar = item.querySelector('.bar');
        let progress = 0;
        const timer = setInterval(() => {
            progress += Math.random() * 30;
            if (progress >= 100) {
                progress = 100;
                clearInterval(timer);
                item.querySelector('.file-size').textContent += ' ✓ 完成';
            }
            bar.style.width = progress + '%';
        }, 300);
    }

    /** 获取待上传的文件列表（数据获取接口） */
    getFiles() {
        return this._fileStore.filter((_, i) => {
            // 排除已从 DOM 中删除的
            return !!this.list.children[i];
        });
    }
}

    document.getElementById('fileInput'),
    document.getElementById('uploadFileList')
);

JavaScript — 级联选择器交互逻辑 
// === 级联选择器交互（纯 JS） ===
// 数据获取方式：调用 getValue() → { path: string[], text: string }
// 数据变化时：仅修改 className，不替换 DOM，界面稳定无闪烁
const cascader = (() => {
    const resultEl = document.getElementById('casResult');

    function updatePath() {
        const cols = document.querySelectorAll('#cascaderDemo .cascader-col');
        let path = [];
        cols.forEach(col => {
            const active = col.querySelector('.cas-item.active');
            if (active) path.push(active.textContent.replace(/[›\s📁]+/g,'').trim());
        });
        resultEl.textContent = '已选：' + path.join(' / ');
        return path;
    }

    document.querySelectorAll('#cascaderDemo .cascader-col .cas-item').forEach(item => {
        item.addEventListener('click', function() {
            // 只修改 className，不重建 DOM
            this.parentElement.querySelectorAll('.cas-item').forEach(i => i.classList.remove('active'));
            this.classList.add('active');
            updatePath();
        });
    });

    /** 获取当前选中路径（数据获取接口） */
    function getValue() {
        const path = updatePath();
        return { path, text: path.join(' / ') };
    }

    return { getValue };
})();
// 获取选中值：cascader.getValue()  →  { path: ['中国大陆','广东省','广州市'], text: '中国大陆 / 广东省 / 广州市' }
