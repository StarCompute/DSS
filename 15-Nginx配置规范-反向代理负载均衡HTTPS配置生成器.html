🔄一、反向代理规范

基础反向代理配置

nginx.conf — 反向代理模板
# ===== 上游服务器定义 =====
upstream backend {
    # 负载均衡策略（默认轮询）
    server 192.168.1.10:8080 weight=3 max_fails=3 fail_timeout=30s;
    server 192.168.1.11:8080 weight=1;
    server 192.168.1.12:8080 backup;  # 热备
    keepalive 32;  # 保持长连接数
}

server {
    listen 80;
    server_name example.com www.example.com;

    # 访问日志（按域名分离）
    access_log /var/log/nginx/example.access.log main;
    error_log  /var/log/nginx/example.error.log warn;

    # 客户端请求体限制
    client_max_body_size 10m;
    client_body_buffer_size 128k;

    # 超时设置
    proxy_connect_timeout 60s;
    proxy_send_timeout    60s;
    proxy_read_timeout    60s;

    # 代理头信息（传递真实 IP）
    proxy_set_header Host              $host;
    proxy_set_header X-Real-IP         $remote_addr;
    proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;

    # 反向代理到后端
    location / {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
    }

    # 静态资源（Nginx 直接响应）
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff2?|ttf|eot)$ {
        root /var/www/static;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}

负载均衡策略

策略指令场景
轮询（默认）默认通用，节点性能一致
加权轮询weight=N节点性能差异较大
IP 哈希ip_hash;需要会话保持 (Session)
最少连接least_conn;长连接场景（WebSocket）
一致性哈希hash $request_uri consistent;缓存服务器集群

2.2 acme.sh 方案（⭐ 推荐）

acme.sh 是一个纯 Shell 脚本实现的 ACME 客户端，零依赖，支持多种 DNS API 自动验证，是目前最灵活的免费证书方案。

2.4 Nginx 完整 SSL 配置（与免费证书整合）

以下为推荐的生产级 HTTPS 配置，与 acme.sh/Certbot 免费证书方案无缝整合。

ssl.conf — 生产级 HTTPS 配置（含 ACME 验证 + HTTP 跳转）
# ===== HTTP → HTTPS 跳转（同时保留 ACME 验证路径）=====
server {
    listen 80;
    server_name example.com www.example.com;

    # ACME 证书验证路径（Let's Encrypt / acme.sh 通用）
    location ^~ /.well-known/acme-challenge/ {
        root /var/www/acme;
        default_type text/plain;
        allow all;
    }

    # 其他请求 301 跳转到 HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    server_name example.com www.example.com;

    # ===== SSL 证书（acme.sh 安装路径）=====
    ssl_certificate     /etc/nginx/ssl/example.com/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/example.com/key.pem;

    # 或使用 Certbot 默认路径：
    # ssl_certificate     /etc/letsencrypt/live/example.com/fullchain.pem;
    # ssl_certificate_key /etc/letsencrypt/live/example.com/privkey.pem;

    # ===== SSL 协议与加密套件（Mozilla Intermediate 推荐）=====
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;  # TLS 1.3 时 off 更安全

    # ===== SSL 会话缓存 =====
    ssl_session_cache   shared:SSL:50m;
    ssl_session_timeout 1d;
    ssl_session_tickets off;

    # ===== OCSP Stapling（加速证书验证）=====
    ssl_stapling on;
    ssl_stapling_verify on;
    resolver 8.8.8.8 8.8.4.4 valid=300s;
    resolver_timeout 5s;

    # ===== HSTS（强制 HTTPS，有效期 2 年）=====
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;

    # ===== 安全响应头 =====
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # 反向代理
    location / {
        proxy_pass http://backend;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
    }
}

2.5 一键部署完整流程（从零到 HTTPS）

完整部署脚本 — acme.sh + Nginx 一条龙
#!/bin/bash
# ===== 从零部署 HTTPS 完整流程 =====
# 使用方法: chmod +x deploy-ssl.sh && ./deploy-ssl.sh example.com www.example.com

DOMAIN="$1"
ALIASES="${@:2}"
ACME_ROOT="/var/www/acme"
SSL_DIR="/etc/nginx/ssl/$DOMAIN"
NGINX_CONF="/etc/nginx/conf.d/$DOMAIN.conf"

# 1. 安装 acme.sh
if [ ! -f "$HOME/.acme.sh/acme.sh" ]; then
    echo "[1/5] 安装 acme.sh..."
    curl https://get.acme.sh | sh -s email="admin@$DOMAIN"
    source ~/.bashrc
else
    echo "[1/5] acme.sh 已安装"
fi

# 2. 创建验证目录
echo "[2/5] 创建 ACME 验证目录..."
mkdir -p "$ACME_ROOT/.well-known/acme-challenge"
mkdir -p "$SSL_DIR"

# 3. 签发证书（Webroot 模式）
echo "[3/5] 签发 SSL 证书..."
~/.acme.sh/acme.sh --issue \
    -d "$DOMAIN" \
    -d "www.$DOMAIN" \
    -w "$ACME_ROOT" \
    --keylength ec-256  # ECC 证书，性能更优

# 4. 安装证书
echo "[4/5] 安装证书到 Nginx..."
~/.acme.sh/acme.sh --install-cert -d "$DOMAIN" \
    --key-file       "$SSL_DIR/key.pem" \
    --fullchain-file "$SSL_DIR/fullchain.pem" \
    --reloadcmd      "nginx -s reload"

# 5. 验证 Nginx 并重载
echo "[5/5] 验证 Nginx 配置并重载..."
nginx -t && nginx -s reload

echo ""
echo "✅ HTTPS 部署完成！"
echo "   证书路径: $SSL_DIR/fullchain.pem"
echo "   密钥路径: $SSL_DIR/key.pem"
echo "   续期命令: 已自动配置 crontab，每日检查"
echo "   手动续期: acme.sh --cron"
echo "   测试评级: https://www.ssllabs.com/ssltest/analyze.html?d=$DOMAIN"

2.6 多站点证书批量管理

批量管理脚本 — 多域名证书统一管理
#!/bin/bash
# ===== 批量签发/续期证书 =====

# 域名列表（每行一个域名）
DOMAINS=(
    "example.com"
    "api.example.com"
    "admin.example.com"
    "static.example.com"
)

for domain in "${DOMAINS[@]}"; do
    echo "处理: $domain"
    
    # 签发证书（DNS API 模式，适合无 80 端口场景）
    ~/.acme.sh/acme.sh --issue \
        --dns dns_dp \
        -d "$domain" \
        -d "*.$domain" \
        --keylength ec-256
    
    # 安装证书
    ~/.acme.sh/acme.sh --install-cert -d "$domain" \
        --key-file       "/etc/nginx/ssl/$domain/key.pem" \
        --fullchain-file "/etc/nginx/ssl/$domain/fullchain.pem" \
        --reloadcmd      "nginx -s reload"
done

echo "✅ 全部证书处理完成"
acme.sh --list

🛡️三、安全头与防护配置

security.conf — 安全头集合
# ===== 安全响应头（建议所有站点统一配置）=====

# 1. 防止点击劫持
add_header X-Frame-Options "SAMEORIGIN" always;

# 2. 防止 MIME 类型嗅探
add_header X-Content-Type-Options "nosniff" always;

# 3. XSS 过滤器
add_header X-XSS-Protection "1; mode=block" always;

# 4. 内容安全策略 CSP（按需调整）
add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' https://cdn.example.com; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:; font-src 'self' data:; connect-src 'self'; frame-ancestors 'self';" always;

# 5. 引用策略
add_header Referrer-Policy "strict-origin-when-cross-origin" always;

# 6. 权限策略
add_header Permissions-Policy "camera=(), microphone=(), geolocation=()" always;

# ===== 限流（防止 CC 攻击）=====
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;
limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

location /api/ {
    limit_req zone=api_limit burst=20 nodelay;
    limit_conn conn_limit 10;
    proxy_pass http://backend;
}

# ===== 限制请求方法 =====
if ($request_method !~ ^(GET|HEAD|POST|PUT|DELETE|PATCH|OPTIONS)$) {
    return 405;
}

🚫四、访问控制：禁止与允许文件/目录访问

⚠️ 安全必备：合理配置访问控制可以有效防止源码泄露、敏感信息暴露、目录遍历等安全风险。 以下提供禁止模式（Deny）和允许模式（Allow）两种配置方案，以及一键复制功能。

访问控制分为两种策略：黑名单模式（禁止特定文件/目录，其余放行）和白名单模式（仅允许特定文件/目录，其余禁止）。根据安全需求选择合适策略。

4.1 禁止特定类型文件访问

通过正则匹配文件后缀，禁止常见敏感文件被外部访问。适用于全局或 server 块。

block-file-types.conf — 按文件类型禁止访问
# ===== 禁止特定后缀文件访问 =====
# 以下配置阻止所有敏感/系统文件被直接下载

location ~* \.(bak|backup|old|orig|save|swp|tmp|temp)$ {
    deny all;
    return 404;  # 返回 404 而非 403，隐藏文件存在信息
}

location ~* \.(sql|sqlite|sqlite3|db|mdb|sql.gz)$ {
    deny all;
    return 404;
}

location ~* \.(git|svn|hg|bzr|cvs)$ {
    deny all;
    return 404;
}

location ~* \.(env|env\.\w+|\.env)$ {
    deny all;
    return 404;
}

location ~* \.(yml|yaml|toml|ini|cfg|conf|config)$ {
    deny all;
    return 404;
}

location ~* \.(log|log\.\d+|access_log|error_log)$ {
    deny all;
    return 404;
}

location ~* \.(md|rst|txt|wiki)$ {
    deny all;
    return 404;
}

location ~* \.(pem|key|crt|csr|p12|pfx|jks|keystore)$ {
    deny all;
    return 404;
}

location ~* \.(DS_Store|htaccess|htpasswd|composer\.lock|package-lock\.json|yarn\.lock)$ {
    deny all;
    return 404;
}

# 禁止访问隐藏文件（以 . 开头）
location ~ /\. {
    deny all;
    return 404;
    access_log off;
    log_not_found off;
}

4.2 禁止特定目录访问

对整个目录设置拒绝访问，适用于需要保护的后台目录、数据目录、源码目录等。

block-directories.conf — 按目录禁止访问
# ===== 方案一：精确匹配目录（推荐）=====
# 禁止访问 .git 目录及所有子内容

    deny all;
    return 404;
}

# 禁止访问 .svn 目录
location ~ /\.svn {
    deny all;
    return 404;
}

# 禁止访问 .idea / .vscode 等 IDE 配置目录
location ~ /\.(idea|vscode|project|settings) {
    deny all;
    return 404;
}

# 禁止访问 node_modules
location ~ /node_modules {
    deny all;
    return 404;
}

# 禁止访问 vendor（PHP 依赖目录）

    deny all;

4.6 白名单模式（Allow）— 仅允许特定路径访问

在某些高安全场景下，推荐采用默认拒绝 + 显式允许的白名单策略。以下提供完整的白名单配置方案。

❌ 黑名单模式（默认放行）
默认：所有请求放行
策略：逐个列出禁止项
风险：遗漏新的敏感路径
依赖持续维护黑名单，易遗漏
✅ 白名单模式（默认拒绝）
默认：所有请求拒绝
策略：逐个列出允许项
安全：新路径自动受保护
默认安全，适合高安全场景

白名单 — 仅允许访问特定文件类型

白名单 vs 黑名单 策略对比

维度黑名单（Deny List）白名单（Allow List）
默认行为放行所有拒绝所有
配置复杂度低（只列出敏感项）高（需列出所有允许项）
安全性中等（依赖维护）高（默认安全）
维护成本低（新增敏感文件需加规则）中（新增功能需加允许规则）
误拦截风险低高（忘记添加允许导致功能异常）
适用场景通用 Web 站点、CMS高安全 API、静态资源 CDN、管理后台
推荐指数⭐⭐⭐⭐（通用场景）⭐⭐⭐（高安全场景）
📋 推荐策略：通用 Web 站点使用黑名单模式（禁止敏感文件 + 隐藏目录）， 高安全场景（API 网关、金融系统）使用白名单模式（默认拒绝 + 显式允许）。 两种模式可以组合使用，例如全局用黑名单，管理后台用白名单限制 IP。

server-optimization.conf — Server 块优化
server {
    listen 443 ssl http2;

        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;

    # 普通静态资源（30 天缓存）
    location ~* \.(js|css|png|jpg|jpeg|gif|svg|ico|woff2?|ttf|eot)$ {
        expires 30d;
        add_header Cache-Control "public, must-revalidate";
        add_header Vary Accept-Encoding;
        access_log off;
    }

    # HTML 不缓存
    location ~* \.html$ {
        expires -1;
        add_header Cache-Control "no-cache, no-store, must-revalidate";
    }

    # ===== 代理缓存示例 =====
    location /api/cached/ {
        proxy_pass http://backend;
        proxy_cache proxy_cache;
        proxy_cache_valid 200 302 10m;
        proxy_cache_valid 404 1m;
        proxy_cache_key "$scheme$request_method$host$request_uri";
        proxy_cache_bypass $http_cache_control;
        add_header X-Cache-Status $upstream_cache_status;
    }

    # ===== 代理缓冲区优化 =====
    location / {
        proxy_pass http://backend;
        proxy_buffering on;
        proxy_buffer_size           16k;
        proxy_buffers               8 16k;
        proxy_busy_buffers_size     32k;
        proxy_temp_file_write_size  32k;
        proxy_read_timeout          60s;
        proxy_connect_timeout       30s;
        proxy_send_timeout          60s;
    }
}
