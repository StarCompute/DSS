🔧一、自定义 DBHelper 设计

核心原则：以下是一个生产级 PHP PDO 封装类，涵盖连接管理、参数化查询、事务、批量操作和日志。 可作为项目中直接引用的通用数据库操作层。支持 PHP、Python、C#、Java 四语言版本。
DBHelper.php — 完整 DBHelper 实现
<?php
declare(strict_types=1);

namespace App\Database;

use PDO;
use PDOException;
use PDOStatement;
use RuntimeException;
use InvalidArgumentException;

/**
 * 数据库操作助手 — PDO 封装。
 *
 * 特性：
 * - 单例连接管理（支持读写分离扩展）
 * - 全部参数化查询，杜绝 SQL 注入
 * - 自动事务 with 回滚
 * - 批量插入优化
 * - 查询日志（调试模式）
 */
class DBHelper
{
    private static ?DBHelper $instance = null;
    private PDO $pdo;
    private bool $debug = false;
    private array $queryLog = [];

    // ─── 配置 ───
    private string $host;
    private int $port;
    private string $dbName;
    private string $user;
    private string $pass;
    private string $charset;

    private function __construct(array $config = [])
    {
        $this->host    = $config['host']    ?? '127.0.0.1';
        $this->port    = $config['port']    ?? 3306;
        $this->dbName  = $config['dbname']  ?? 'test';
        $this->user    = $config['user']    ?? 'root';
        $this->pass    = $config['pass']    ?? '';
        $this->charset = $config['charset'] ?? 'utf8mb4';
        $this->debug   = $config['debug']   ?? false;
        $this->connect();
    }

    // ─── 单例入口 ───
    public static function getInstance(array $config = []): self
    {
        if (self::$instance === null) {
            self::$instance = new self($config);
        }
        return self::$instance;
    }

    // ─── 连接 ───
    private function connect(): void
    {
        $dsn = sprintf(
            'mysql:host=%s;port=%d;dbname=%s;charset=%s',
            $this->host, $this->port, $this->dbName, $this->charset
        );
        $options = [
            PDO::ATTR_ERRMODE            => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES   => false,
            PDO::MYSQL_ATTR_INIT_COMMAND => "SET NAMES {$this->charset}",
        ];
        try {
            $this->pdo = new PDO($dsn, $this->user, $this->pass, $options);
        } catch (PDOException $e) {
            throw new RuntimeException("数据库连接失败: " . $e->getMessage(), 500);
        }
    }

    // ─── 查询：返回多行 ───
    public function select(string $sql, array $params = []): array
    {
        return $this->execute($sql, $params)->fetchAll();
    }

    // ─── 查询：返回单行 ───
    public function selectOne(string $sql, array $params = []): ?array
    {
        $row = $this->execute($sql, $params)->fetch();
        return $row !== false ? $row : null;
    }

    // ─── 查询：返回单个值 ───
    public function selectValue(string $sql, array $params = []): mixed
    {
        $row = $this->execute($sql, $params)->fetch(PDO::FETCH_NUM);
        return $row !== false ? $row[0] : null;
    }

    // ─── 执行写操作 (INSERT/UPDATE/DELETE) ───
    public function execute(string $sql, array $params = []): PDOStatement
    {
        $start = microtime(true);
        try {
            $stmt = $this->pdo->prepare($sql);
            $stmt->execute($params);
        } catch (PDOException $e) {
            throw new RuntimeException(
                "SQL 执行失败: " . $e->getMessage() . " | SQL: " . $sql,
                (int)$e->getCode()
            );
        }
        if ($this->debug) {
            $this->queryLog[] = [
                'sql'    => $sql,
                'params' => $params,
                'time'   => round((microtime(true) - $start) * 1000, 2) . 'ms',
            ];
        }
        return $stmt;
    }

    // ─── 获取最后插入 ID ───
    public function lastInsertId(): string { return $this->pdo->lastInsertId(); }

    // ─── 批量插入 ───
    public function batchInsert(string $table, array $rows): int
    {
        if (empty($rows)) return 0;
        $columns = array_keys(reset($rows));
        $placeholders = '(' . implode(',', array_fill(0, count($columns), '?')) . ')';
        $allPlaceholders = implode(',', array_fill(0, count($rows), $placeholders));
        $sql = sprintf('INSERT IGNORE INTO `%s` (`%s`) VALUES %s', $table, implode('`,`', $columns), $allPlaceholders);
        $params = [];
        foreach ($rows as $row) { foreach ($row as $value) { $params[] = $value; } }
        return $this->execute($sql, $params)->rowCount();
    }

    // ─── 事务（回调模式） ───
    public function transaction(callable $callback): mixed
    {
        $this->pdo->beginTransaction();
        try {
            $result = $callback($this);
            $this->pdo->commit();
            return $result;
        } catch (\Throwable $e) {
            $this->pdo->rollBack();
            throw $e;
        }
    }

    // ─── 分页查询 ───
    public function paginate(string $sql, array $params, int $page, int $pageSize = 20): array
    {
        $page = max(1, $page);
        $pageSize = min(max(1, $pageSize), 200);
        $countSql = "SELECT COUNT(*) FROM ({$sql}) AS _tmp";
        $total = (int)$this->selectValue($countSql, $params);
        $offset = ($page - 1) * $pageSize;
        $data = $this->select("{$sql} LIMIT {$pageSize} OFFSET {$offset}", $params);
        return [
            'data' => $data, 'total' => $total,
            'page' => $page, 'page_size' => $pageSize,
            'pages' => (int)ceil($total / $pageSize),
        ];
    }

    // ─── 插入 / 更新 / 删除 ───
    public function insert(string $table, array $data): string
    {
        $columns = array_keys($data);
        $holders = array_fill(0, count($columns), '?');
        $sql = sprintf('INSERT INTO `%s` (`%s`) VALUES (%s)', $table, implode('`,`', $columns), implode(',', $holders));
        $this->execute($sql, array_values($data));
        return $this->lastInsertId();
    }

    public function update(string $table, array $data, string $where, array $whereParams = []): int
    {
        $sets = []; foreach ($data as $col => $v) { $sets[] = "`{$col}` = ?"; }
        $sql = sprintf('UPDATE `%s` SET %s WHERE %s', $table, implode(', ', $sets), $where);
        return $this->execute($sql, array_merge(array_values($data), $whereParams))->rowCount();
    }

    public function delete(string $table, string $where, array $params = []): int
    {
        return $this->execute("DELETE FROM `{$table}` WHERE {$where}", $params)->rowCount();
    }

    // ─── 原子增减 ───
    public function increment(string $table, string $column, string $where, array $params = [], int $amount = 1): int
    {
        $sql = "UPDATE `{$table}` SET `{$column}` = `{$column}` + ? WHERE {$where}";
        return $this->execute($sql, array_merge([$amount], $params))->rowCount();
    }
    public function decrement(string $table, string $column, string $where, array $params = [], int $amount = 1): int
    {
        return $this->increment($table, $column, $where, $params, -$amount);
    }

    // ─── 手动事务控制 ───
    public function beginTransaction(): void { $this->pdo->beginTransaction(); }
    public function commit(): void { $this->pdo->commit(); }
    public function rollBack(): void { if ($this->pdo->inTransaction()) { $this->pdo->rollBack(); } }

    // ─── 判断存在 / 计数 ───
    public function exists(string $tableOrSql, array $params = [], bool $isRawSql = false): bool { /* ... */ }
    public function count(string $tableOrSql, array $params = [], bool $isRawSql = false): int { /* ... */ }

    // ─── 连接检测 ───
    public function ping(): bool { /* ping + auto-reconnect */ return true; }

    // ─── 调试日志 ───
    public function getQueryLog(): array { return $this->queryLog; }

    // 禁止克隆/反序列化
    private function __clone() {}
    public function __wakeup() { throw new RuntimeException("禁止反序列化单例"); }
}
db_helper.py — SQLAlchemy Core + 连接池
"""数据库操作助手 — 基于 SQLAlchemy Core。"""
from __future__ import annotations
from contextlib import contextmanager
from typing import Any, Optional
from sqlalchemy import create_engine, text
from sqlalchemy.engine import Engine
from sqlalchemy.pool import QueuePool

class DBHelper:
    """数据库操作封装。"""
    def __init__(self, url: str = "mysql+pymysql://user:pass@127.0.0.1:3306/db?charset=utf8mb4",
                 pool_size: int = 10, pool_recycle: int = 3600, echo: bool = False) -> None:
        self.engine: Engine = create_engine(url, poolclass=QueuePool,
            pool_size=pool_size, pool_recycle=pool_recycle, pool_pre_ping=True, echo=echo)

    def select(self, sql: str, params: dict | None = None) -> list[dict]:
        with self.engine.connect() as conn:
            rows = conn.execute(text(sql), params or {})
            return [dict(row._mapping) for row in rows]

    def select_one(self, sql: str, params: dict | None = None) -> dict | None:
        with self.engine.connect() as conn:
            row = conn.execute(text(sql), params or {}).fetchone()
            return dict(row._mapping) if row else None

    def select_value(self, sql: str, params: dict | None = None) -> Any:
        with self.engine.connect() as conn:
            return conn.execute(text(sql), params or {}).scalar()

    def execute(self, sql: str, params: dict | None = None) -> int:
        with self.engine.begin() as conn:
            result = conn.execute(text(sql), params or {})
            return result.rowcount

    @contextmanager
    def transaction(self):
        with self.engine.begin() as conn:
            yield conn

    def paginate(self, sql: str, params: dict, page: int, page_size: int = 20) -> dict:
        total = self.select_value(f"SELECT COUNT(*) FROM ({sql}) AS _tmp", params)
        offset = (page - 1) * page_size
        data = self.select(f"{sql} LIMIT :limit OFFSET :offset", {**params, "limit": page_size, "offset": offset})
        return {"data": data, "total": total, "page": page, "page_size": page_size,
                "pages": (total + page_size - 1) // page_size}

    def insert(self, table: str, data: dict) -> int:
        cols = list(data.keys())
        holders = [f":{c}" for c in cols]
        sql = f"INSERT INTO `{table}` (`{'`,`'.join(cols)}`) VALUES ({','.join(holders)})"
        with self.engine.begin() as conn:
            result = conn.execute(text(sql), data)
            return result.lastrowid

    def update(self, table: str, data: dict, where: str, where_params: dict | None = None) -> int:
        sets = [f"`{c}` = :set_{c}" for c in data]
        params = {f"set_{c}": v for c, v in data.items()}
        if where_params: params.update(where_params)
        return self.execute(f"UPDATE `{table}` SET {', '.join(sets)} WHERE {where}", params)

    def delete(self, table: str, where: str, params: dict | None = None) -> int:
        return self.execute(f"DELETE FROM `{table}` WHERE {where}", params or {})

    def increment(self, table: str, column: str, where: str, params: dict | None = None, amount: int = 1) -> int:
        p = params or {}; p["_amt"] = amount
        return self.execute(f"UPDATE `{table}` SET `{column}` = `{column}` + :_amt WHERE {where}", p)
DbHelper.cs — Dapper 封装
using System.Data;
using Dapper;
using MySql.Data.MySqlClient;

namespace Company.Project.Data;

public class DbHelper : IDisposable
{
    private readonly MySqlConnection _conn;

    public DbHelper(string connectionString) { _conn = new MySqlConnection(connectionString); _conn.Open(); }

    public async Task<IEnumerable<T>> SelectAsync<T>(string sql, object? param = null)
        => await _conn.QueryAsync<T>(sql, param);

    public async Task<T?> SelectOneAsync<T>(string sql, object? param = null)
        => await _conn.QueryFirstOrDefaultAsync<T>(sql, param);

    public async Task<int> ExecuteAsync(string sql, object? param = null, IDbTransaction? tx = null)
        => await _conn.ExecuteAsync(sql, param, tx);

    public async Task<(IEnumerable<T> Data, int Total)> PaginateAsync<T>(string sql, object param, int page, int pageSize = 20)
    {
        var total = await _conn.ExecuteScalarAsync<int>($"SELECT COUNT(*) FROM ({sql}) AS _tmp", param);
        var offset = (page - 1) * pageSize;
        var data = await _conn.QueryAsync<T>($"{sql} LIMIT {pageSize} OFFSET {offset}", param);
        return (data, total);
    }

    public IDbTransaction BeginTransaction() => _conn.BeginTransaction();

    public async Task<bool> ExistsAsync(string sql, object? param = null)
        => await _conn.ExecuteScalarAsync<int>($"SELECT EXISTS({sql})", param) == 1;

    public async Task<long> InsertAsync(string table, object data)
    {
        var props = data.GetType().GetProperties().Where(p => p.GetValue(data) != null).ToList();
        var cols = string.Join("`,`", props.Select(p => p.Name));
        var holders = string.Join(",", props.Select(p => $"@{p.Name}"));
        return await _conn.ExecuteScalarAsync<long>(
            $"INSERT INTO `{table}` (`{cols}`) VALUES ({holders}); SELECT LAST_INSERT_ID()", data);
    }

    public async Task<int> DeleteAsync(string table, string where, object? param = null)
        => await _conn.ExecuteAsync($"DELETE FROM `{table}` WHERE {where}", param);

    public async Task<int> IncrementAsync(string table, string column, string where,
        object? param = null, int amount = 1)
        => await _conn.ExecuteAsync(
            $"UPDATE `{table}` SET `{column}` = `{column}` + {amount} WHERE {where}", param);

    public void Dispose() => _conn?.Dispose();
}
DbHelper.java — JdbcTemplate 封装
package com.company.project.data;

import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.jdbc.core.RowMapper;
import org.springframework.stereotype.Component;
import java.util.*;

@Component
public class DbHelper {
    private final JdbcTemplate jdbc;
    public DbHelper(JdbcTemplate jdbc) { this.jdbc = jdbc; }

    public <T> List<T> select(String sql, RowMapper<T> mapper, Object... params) {
        return jdbc.query(sql, mapper, params);
    }

    public <T> Optional<T> selectOne(String sql, RowMapper<T> mapper, Object... params) {
        List<T> list = jdbc.query(sql, mapper, params);
        return list.isEmpty() ? Optional.empty() : Optional.of(list.get(0));
    }

    public <T> T selectValue(String sql, Class<T> clazz, Object... params) {
        return jdbc.queryForObject(sql, clazz, params);
    }

    public int execute(String sql, Object... params) { return jdbc.update(sql, params); }

    public record PageResult<T>(List<T> data, long total, int page, int pageSize) {}

    public <T> PageResult<T> paginate(String sql, RowMapper<T> mapper, int page, int pageSize, Object... params) {
        Long total = jdbc.queryForObject("SELECT COUNT(*) FROM (" + sql + ") AS _tmp", Long.class, params);
        int offset = (page - 1) * pageSize;
        List<T> data = jdbc.query(sql + " LIMIT " + pageSize + " OFFSET " + offset, mapper, params);
        return new PageResult<>(data, total, page, pageSize);
    }

    public long insert(String table, Map<String, Object> data) { /* SimpleJdbcInsert */ return 0L; }
    public int delete(String table, String where, Object... params) {
        return jdbc.update("DELETE FROM `" + table + "` WHERE " + where, params);
    }
    public int increment(String table, String column, String where, int amount, Object... params) {
        String sql = "UPDATE `" + table + "` SET `" + column + "` = `" + column + "` + ? WHERE " + where;
        Object[] allParams = new Object[params.length + 1];
        allParams[0] = amount;
        System.arraycopy(params, 0, allParams, 1, params.length);
        return jdbc.update(sql, allParams);
    }
}

DBHelper 核心设计原则

SQL 注入防护

  • 所有查询必须参数化绑定
  • 永不拼接用户输入到 SQL 字符串
  • 表名/列名动态时，使用白名单校验

连接管理

  • 长连接 + 连接池，避免频繁建连
  • 设置连接超时和最大生命周期
  • 连接前检测有效性 (pre_ping)

事务规范

  • 事务尽量短，不放网络 I/O
  • 写操作必须包裹事务
  • 避免长事务导致锁等待

错误处理

  • 统一异常封装，不泄露 SQL 细节
  • 死锁重试（最多 3 次）
  • 慢查询日志 ≥ 200ms

📦二、ORM 框架选型与推荐

核心原则：以下仅收录经过性能验证的高效 ORM / SQL 工具。 全功能重型 ORM（如 Hibernate、Entity Framework 跟踪查询、Eloquent 关联预加载缺失场景） 在复杂查询和批量操作中存在显著的 N+1、内存占用大、生成 SQL 不可控等问题，不做推荐。

💡 选型原则：
  • 简单 CRUD → 轻量 ORM（Dapper / MyBatis-Plus / Peewee）
  • 复杂查询 / 报表 → SQL 构造器 / Mapper（SQLAlchemy Core / jOOQ / DBHelper）
  • 高并发 / 常驻内存 → 无状态 SQL 封装，避免 ORM Unit of Work
  • 所有场景 → 禁止使用全功能重型 ORM 的默认跟踪/懒加载模式
以上推荐的框架均使用 MIT / Apache-2.0 许可证，可免费商业使用。

🏗️三、仓储模式（Repository Pattern）

核心原则：仓储模式将数据访问逻辑封装在独立层，使业务逻辑与持久化细节解耦。 通过接口定义契约，实现可替换性，便于单元测试。

PHP 仓储模式示例

UserRepository.php — 接口 + 实现 + Service 调用
// ============ 接口 ============
interface UserRepositoryInterface
{
    public function findById(int $id): ?User;
    public function findByEmail(string $email): ?User;
    public function search(UserQuery $query): array;
    public function save(User $user): User;
    public function delete(int $id): bool;
}

// ============ 实现 ============
class UserRepository implements UserRepositoryInterface
{
    public function __construct(private DBHelper $db) {}

    public function findById(int $id): ?User
    {
        $row = $this->db->selectOne(
            'SELECT * FROM users WHERE id = ? AND deleted_at IS NULL', [$id]
        );
        return $row ? User::fromArray($row) : null;
    }

    public function save(User $user): User
    {
        if ($user->id) {
            $this->db->execute(
                'UPDATE users SET name=?, email=?, updated_at=NOW() WHERE id=?',
                [$user->name, $user->email, $user->id]
            );
        } else {
            $this->db->execute(
                'INSERT INTO users (name, email, created_at, updated_at) VALUES (?, ?, NOW(), NOW())',
                [$user->name, $user->email]
            );
            $user->id = (int)$this->db->lastInsertId();
        }
        return $user;
    }

    public function delete(int $id): bool
    {
        // 软删除
        return $this->db->execute(
            'UPDATE users SET deleted_at = NOW() WHERE id = ?', [$id]
        )->rowCount() > 0;
    }
}

// ============ Service 层使用 ============
class UserService
{
    public function __construct(private UserRepositoryInterface $userRepo) {}

    public function activateUser(int $id): User
    {
        $user = $this->userRepo->findById($id)
            ?? throw new NotFoundException("用户不存在");
        $user->status = 'active';
        return $this->userRepo->save($user);
    }
}

🛡️四、SQL 安全与性能规范

核心原则：SQL 注入是 OWASP Top 10 高风险漏洞。 所有用户输入必须通过参数化绑定进入 SQL，动态表名/列名/排序字段必须通过白名单校验。

SQL 注入对照

场景❌ 危险写法✅ 安全写法
条件查询"SELECT * FROM users WHERE id = $id"SELECT * FROM users WHERE id = ?
LIKE 查询"WHERE name LIKE '%$kw%'"WHERE name LIKE CONCAT('%', ?, '%')
IN 查询"WHERE id IN ($ids)"WHERE id IN (?, ?, ?) [动态占位符]
排序字段"ORDER BY $col $dir"白名单校验 + 硬编码字段名
表名动态"SELECT * FROM $table"白名单 + 反引号包裹

查询优化要点

索引原则

  • WHERE / JOIN / ORDER BY 列建索引
  • 复合索引遵循最左前缀
  • 避免在索引列上使用函数
  • 用 EXPLAIN 分析执行计划

查询优化

  • SELECT 指定列名，不用 SELECT *
  • 大偏移分页用游标/延迟关联
  • 批量操作分批次（每批 ≤1000）
  • 避免在循环中执行 SQL (N+1)

高危操作

  • 生产环境禁止 DROP/TRUNCATE
  • UPDATE/DELETE 必须有 WHERE
  • 大表 DDL 用 pt-online-schema-change
  • 慢查询阈值设为 200ms

连接规范

  • 连接池最小 5，最大 20~50
  • 连接超时 30s，等待超时 3s
  • 读写分离：写主库，读从库
  • 设置 sql_mode 严格模式

✅六、数据库操作 · 检查清单

✅ 是否所有 SQL 查询都使用参数化绑定，无字符串拼接用户输入？
✅ 动态表名/列名/排序字段是否经过白名单校验？
✅ DBHelper 是否使用单例模式管理连接？
✅ PDO 是否设置 ERRMODE_EXCEPTION + EMULATE_PREPARES=false？
✅ 是否提供了批量插入（batchInsert）方法？
✅ 事务是否同时提供回调模式和手动控制模式？
✅ 是否实现了分页查询（paginate），含 total/page/pages？
✅ 是否提供了原子增减（increment/decrement）方法？
✅ 连接池是否配置了 最小连接数、最大连接数、超时？
✅ 是否选用了高性能轻量 ORM（如 Dapper/MyBatis/SQLAlchemy Core）？
✅ 是否避免了全功能重型 ORM 的默认跟踪/懒加载模式？
✅ 生产环境是否禁止了 DROP/TRUNCATE 操作？
✅ UPDATE/DELETE 语句是否必须带 WHERE？
✅ 大偏移分页是否使用游标/延迟关联替代 OFFSET？
✅ 批量操作是否分批进行（每批 ≤ 1000）？
✅ 是否配置了慢查询日志（阈值 200ms）？
✅ 数据访问层是否通过仓储模式与业务逻辑解耦？
✅ 错误信息是否不暴露 SQL 细节给前端？
