遵循 PSR-1/PSR-12 命名风格，PDO 参数化查询 + 自建 DBHelper 封装。完整代码见 🗃️ 数据库操作规范。

DBHelper + UserRepository — PHP PDO 标准写法
// === DBHelper 核心（PDO 单例 + 全参数化）===
class DBHelper
{
    private static ?DBHelper $instance = null;
    private PDO $pdo;

    private function __construct(array $cfg) {
        $dsn = sprintf('mysql:host=%s;port=%d;dbname=%s;charset=utf8mb4', $cfg['host']??'127.0.0.1', $cfg['port']??3306, $cfg['dbname']);
        $this->pdo = new PDO($dsn, $cfg['user'], $cfg['pass'], [
            PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES => false,   // 真预处理，防注入
        ]);
    }
    public static function getInstance(array $cfg = []): self {
        return self::$instance ??= new self($cfg);
    }
    public function select(string $sql, array $params = []): array {
        $stmt = $this->pdo->prepare($sql); $stmt->execute($params); return $stmt->fetchAll();
    }
    public function selectOne(string $sql, array $params = []): ?array {
        $stmt = $this->pdo->prepare($sql); $stmt->execute($params); $row = $stmt->fetch(); return $row ?: null;
    }
    public function selectValue(string $sql, array $params = []): mixed {
        $row = $this->pdo->prepare($sql); $row->execute($params); $r = $row->fetch(PDO::FETCH_NUM); return $r[0] ?? null;
    }
    public function execute(string $sql, array $params = []): int {
        $stmt = $this->pdo->prepare($sql); $stmt->execute($params); return $stmt->rowCount();
    }
    public function lastInsertId(): string { return $this->pdo->lastInsertId(); }
    public function exists(string $sql, array $params = []): bool {
        return $this->selectValue("SELECT EXISTS($sql)", $params) == 1;
    }
    public function transaction(callable $cb): mixed {
        $this->pdo->beginTransaction();
        try { $r = $cb($this); $this->pdo->commit(); return $r; }
        catch (\Throwable $e) { $this->pdo->rollBack(); throw $e; }
    }
    public function beginTransaction(): void { $this->pdo->beginTransaction(); }
    public function commit(): void { $this->pdo->commit(); }
    public function rollBack(): void { $this->pdo->rollBack(); }
    private function __clone() {}
}

// === Repository 使用 DBHelper（接口 + 实现）===
interface UserRepositoryInterface {
    public function findById(int $id): ?User;
    public function findByEmail(string $email): ?User;
    public function save(User $user): User;
    public function delete(int $id): bool;
}
class UserRepository implements UserRepositoryInterface
{
    public function __construct(private DBHelper $db) {}
    
    public function findById(int $id): ?User {
        $row = $this->db->selectOne('SELECT * FROM users WHERE id=? AND deleted_at IS NULL', [$id]);
        return $row ? User::fromArray($row) : null;
    }
    public function findByEmail(string $email): ?User {
        $row = $this->db->selectOne('SELECT * FROM users WHERE email=? AND deleted_at IS NULL', [$email]);
        return $row ? User::fromArray($row) : null;
    }
    public function save(User $user): User {
        if ($user->id) {
            $this->db->execute('UPDATE users SET name=?,email=?,updated_at=NOW() WHERE id=?',
                [$user->name, $user->email, $user->id]);
        } else {
            $this->db->execute('INSERT INTO users (name,email,created_at,updated_at) VALUES (?,?,NOW(),NOW())',
                [$user->name, $user->email]);
            $user->id = (int)$this->db->lastInsertId();
        }
        return $user;
    }
    public function delete(int $id): bool {
        return $this->db->execute('UPDATE users SET deleted_at=NOW() WHERE id=?', [$id]) > 0;
    }
}

遵循 PEP 8 snake_case 风格，使用 SQLAlchemy Core 或驱动参数化。完整代码见 🗃️ 数据库操作规范。

db_helper.py + user_repo.py — SQLAlchemy Core 标准写法
# === db_helper.py — SQLAlchemy Core + 连接池 ===
from __future__ import annotations
from typing import Any
from sqlalchemy import create_engine, text
from sqlalchemy.engine import Engine
from sqlalchemy.pool import QueuePool

class DBHelper:
    def __init__(self, url: str = "mysql+pymysql://user:pass@127.0.0.1:3306/db?charset=utf8mb4",
                 pool_size: int = 10, echo: bool = False) -> None:
        self.engine: Engine = create_engine(
            url, poolclass=QueuePool, pool_size=pool_size,
            pool_recycle=3600, pool_pre_ping=True, echo=echo)

    def select(self, sql: str, params: dict | None = None) -> list[dict]:
        with self.engine.connect() as conn:
            rows = conn.execute(text(sql), params or {})
            return [dict(row._mapping) for row in rows]

    def select_one(self, sql: str, params: dict | None = None) -> dict | None:
        with self.engine.connect() as conn:
            row = conn.execute(text(sql), params or {}).fetchone()
            return dict(row._mapping) if row else None

    def select_value(self, sql: str, params: dict | None = None) -> Any:
        with self.engine.connect() as conn:
            return conn.execute(text(sql), params or {}).scalar()

    def execute(self, sql: str, params: dict | None = None) -> int:
        with self.engine.begin() as conn:
            return conn.execute(text(sql), params or {}).rowcount

    def exists(self, sql: str, params: dict | None = None) -> bool:
        return self.select_value(f"SELECT EXISTS({sql})", params) == 1

    def paginate(self, sql: str, params: dict, page: int, page_size: int = 20) -> dict:
        total = self.select_value(f"SELECT COUNT(*) FROM ({sql}) AS _tmp", params)
        offset = (page - 1) * page_size
        data = self.select(f"{sql} LIMIT :limit OFFSET :offset",
                           {**params, "limit": page_size, "offset": offset})
        return {"data": data, "total": total, "page": page, "page_size": page_size}

# === user_repo.py — 仓储模式（snake_case）===
from app.models.user import User

class UserRepository:
    def __init__(self, db: DBHelper) -> None:
        self._db = db

    def find_by_id(self, user_id: int) -> User | None:
        row = self._db.select_one(
            "SELECT * FROM users WHERE id=:uid AND deleted_at IS NULL", {"uid": user_id})
        return User(**row) if row else None

    def find_by_email(self, email: str) -> User | None:
        row = self._db.select_one(
            "SELECT * FROM users WHERE email=:email AND deleted_at IS NULL",
            {"email": email.lower()})
        return User(**row) if row else None

    def save(self, user: User) -> User:
        if user.id:
            self._db.execute(
                "UPDATE users SET name=:n,email=:e,updated_at=NOW() WHERE id=:id",
                {"n": user.name, "e": user.email, "id": user.id})
        else:
            self._db.execute(
                "INSERT INTO users (name,email) VALUES (:n,:e)",
                {"n": user.name, "e": user.email})
        return user

    def soft_delete(self, user_id: int) -> bool:
        return self._db.execute(
            "UPDATE users SET deleted_at=NOW() WHERE id=:id", {"id": user_id}) > 0

DbHelper.cs + UserRepository.cs — Dapper 标准写法
using System.Data;
using Dapper;
using MySql.Data.MySqlClient;

namespace Company.Project.Data;

// === DbHelper（Dapper 封装） ===
public class DbHelper : IDisposable
{
    private readonly MySqlConnection _conn;
    public DbHelper(string connectionString) {
        _conn = new MySqlConnection(connectionString);
    }

    public async Task OpenAsync() => await _conn.OpenAsync();

    // 查询列表
    public async Task<IEnumerable<T>> SelectAsync<T>(string sql, object? param = null)
        => await _conn.QueryAsync<T>(sql, param);

    // 查询单条
    public async Task<T?> SelectOneAsync<T>(string sql, object? param = null)
        => await _conn.QueryFirstOrDefaultAsync<T>(sql, param);

    // 查询单个值
    public async Task<T> SelectValueAsync<T>(string sql, object? param = null)
        => await _conn.ExecuteScalarAsync<T>(sql, param);

    // 执行写操作
    public async Task<int> ExecuteAsync(string sql, object? param = null, IDbTransaction? tx = null)
        => await _conn.ExecuteAsync(sql, param, tx);

    // 判断是否存在
    public async Task<bool> ExistsAsync(string sql, object? param = null)
        => await _conn.ExecuteScalarAsync<int>($"SELECT EXISTS({sql})", param) == 1;

    // 分页
    public async Task<(IEnumerable<T> Data, int Total)> PaginateAsync<T>(
        string sql, object param, int page, int pageSize = 20) {
        var total = await _conn.ExecuteScalarAsync<int>($"SELECT COUNT(*) FROM ({sql}) _t", param);
        var data = await _conn.QueryAsync<T>($"{sql} LIMIT {pageSize} OFFSET {(page-1)*pageSize}", param);
        return (data, total);
    }

    // 事务
    public IDbTransaction BeginTransaction() => _conn.BeginTransaction();

    public void Dispose() => _conn?.Dispose();
}

// === IUserRepository + 实现 ===
public interface IUserRepository {
    Task<User?> FindByIdAsync(long id, CancellationToken ct = default);
    Task<User> SaveAsync(User user, CancellationToken ct = default);
    Task<bool> SoftDeleteAsync(long id, CancellationToken ct = default);
}

public class UserRepository : IUserRepository
{
    private readonly DbHelper _db;
    public UserRepository(DbHelper db) => _db = db;

    public async Task<User?> FindByIdAsync(long id, CancellationToken ct = default) {
        var sql = "SELECT * FROM users WHERE id=@Id AND deleted_at IS NULL";
        return await _db.SelectOneAsync<User>(sql, new { Id = id });
    }

    public async Task<User> SaveAsync(User user, CancellationToken ct = default) {
        if (user.Id > 0)
            await _db.ExecuteAsync(
                "UPDATE users SET name=@Name,email=@Email,updated_at=NOW() WHERE id=@Id", user);
        else
            user.Id = await _db.SelectValueAsync<long>(
                "INSERT INTO users (name,email,created_at) VALUES (@Name,@Email,NOW()); SELECT LAST_INSERT_ID()", user);
        return user;
    }

    public async Task<bool> SoftDeleteAsync(long id, CancellationToken ct = default)
        => await _db.ExecuteAsync("UPDATE users SET deleted_at=NOW() WHERE id=@Id", new { Id = id }) > 0;
}

import com.company.project.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.Optional;

/**
 * 用户业务逻辑层。
 */
@Slf4j
@Service
@RequiredArgsConstructor
@Transactional(readOnly = true)
public class UserService {

    // 1. 常量
    private static final int CACHE_TTL_MINUTES = 60;

    // 2. 依赖（构造器注入）
    private final UserRepository userRepo;

    /**
     * 按 ID 查找活跃用户。
     */
    public Optional<User> findActiveUser(long userId) {
        if (userId <= 0) {
            throw new IllegalArgumentException("无效的用户 ID: " + userId);
        }
        return userRepo.findByIdAndStatus(userId, UserStatus.ACTIVE);
    }

    /**
     * 创建用户（写事务）。
     */
    @Transactional
    public User createUser(CreateUserRequest request) {
        validateRequest(request);

        User user = User.builder()
                .name(normalizeName(request.getName()))
                .email(request.getEmail().toLowerCase())
                .status(UserStatus.ACTIVE)
                .build();

        User saved = userRepo.save(user);
        log.info("用户创建成功: id={}", saved.getId());
        return saved;
    }

    // 3. 私有方法
    private void validateRequest(CreateUserRequest request) {
        if (request.getName() == null || request.getName().isBlank()) {
            throw new IllegalArgumentException("用户名不能为空");
        }
    }

    private String normalizeName(String name) {
        return name.trim().replaceAll("\\s+", " ");
    }
}

    private final JdbcTemplate jdbc;

    // 查询列表
    public <T> List<T> select(String sql, RowMapper<T> mapper, Object... params) {
        return jdbc.query(sql, mapper, params);
    }

    // 查询单条
    public <T> Optional<T> selectOne(String sql, RowMapper<T> mapper, Object... params) {
        List<T> list = jdbc.query(sql, mapper, params);
        return list.isEmpty() ? Optional.empty() : Optional.of(list.get(0));
    }

    // 查询单个值
    public <T> T selectValue(String sql, Class<T> clazz, Object... params) {
        return jdbc.queryForObject(sql, clazz, params);
    }

    // 执行写操作
    public int execute(String sql, Object... params) {
        return jdbc.update(sql, params);
    }

    // 判断是否存在
    public boolean exists(String sql, Object... params) {
        Integer count = jdbc.queryForObject("SELECT EXISTS(" + sql + ")", Integer.class, params);
        return count != null && count == 1;
    }

    // 分页
    public record PageResult<T>(List<T> data, long total, int page, int pageSize) {}

    public <T> PageResult<T> paginate(String sql, RowMapper<T> mapper,
                                       int page, int pageSize, Object... params) {
        Long total = jdbc.queryForObject("SELECT COUNT(*) FROM (" + sql + ") _t", Long.class, params);
        int offset = (page - 1) * pageSize;
        List<T> data = jdbc.query(sql + " LIMIT " + pageSize + " OFFSET " + offset, mapper, params);
        return new PageResult<>(data, total, page, pageSize);
    }
}

// === UserRepository 接口 + 实现 ===
public interface UserRepository {
    Optional<User> findById(long id);
    User save(User user);
    boolean softDelete(long id);
}

@Component
@RequiredArgsConstructor
public class UserRepositoryImpl implements UserRepository {
    private final DbHelper db;

    private final RowMapper<User> userMapper = (rs, rowNum) -> User.builder()
        .id(rs.getLong("id")).name(rs.getString("name"))
        .email(rs.getString("email")).status(rs.getString("status")).build();

    @Override
    public Optional<User> findById(long id) {
        return db.selectOne(
            "SELECT * FROM users WHERE id=? AND deleted_at IS NULL", userMapper, id);
    }

    @Override
    @Transactional
    public User save(User user) {
        if (user.getId() != null && user.getId() > 0) {
            db.execute("UPDATE users SET name=?,email=?,updated_at=NOW() WHERE id=?",
                user.getName(), user.getEmail(), user.getId());
        } else {
            db.execute("INSERT INTO users (name,email,created_at) VALUES (?,?,NOW())",
                user.getName(), user.getEmail());
        }
        return user;
    }

    @Override
    public boolean softDelete(long id) {
        return db.execute("UPDATE users SET deleted_at=NOW() WHERE id=?", id) > 0;
    }
}

🛠️通用工程规范（跨语言）

架构分层

  • Controller → Service → Repository → DB
  • 每层只依赖下一层，不可反向引用
  • DTO 用于层间数据传输
  • Domain 模型独立于持久化框架

异常处理

  • 业务异常用自定义异常类
  • 全局异常处理器统一捕获
  • 异常消息面向用户可理解
  • 日志记录完整堆栈，API 返回脱敏信息

代码质量

  • 函数不超过 50 行（建议 20 行内）
  • 类不超过 500 行
  • 参数不超过 5 个，多用 DTO 封装
  • 圈复杂度 < 10

安全红线

  • SQL 必须参数化绑定
  • 用户输入必须校验+转义
  • 敏感配置不入库不提交
  • 密码必须哈希（bcrypt/argon2）
